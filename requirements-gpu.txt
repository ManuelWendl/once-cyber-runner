# Additional GPU dependencies on top of the shared r2dreamer_env.
# Install into a PERSONAL conda env cloned from the team env:
#   conda create -y -p $HOME/envs/cyberrunner_gpu \
#       --clone /cluster/courses/pmlr/teams/team06/r2dreamer_env
#   conda activate $HOME/envs/cyberrunner_gpu
#   pip install -r requirements-gpu.txt
#
# Pins: mujoco / mujoco-mjx are locked to 3.3.0 because the existing CPU
# env code (envs/cyberrunner.py::build_model) uses the mujoco 3.3 MjSpec
# API (scalar damping, etc.). Newer mujoco breaks that call site.
# brax 0.10.x is the last series built against mujoco 3.3.x.
#
# jax/jaxlib bumped to 0.4.38 for Blackwell (sm_120, RTX 50-series) support:
# 0.4.30's bundled ptxas targets sm_90a and fails to forward-compile to
# sm_120. 0.4.38 bundles CUDA 12.8 wheels with sm_120 PTX.
# flax bumped to 0.10.2 because 0.8.5 uses the removed `main.level` jax
# internal API that was dropped after jax 0.4.34.
